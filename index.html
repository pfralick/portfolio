                                                var word, wordSplit, wordLength, i, wordReverse="";

                                                //1. Get user input
                                                var word = String($("#inputReverse").val());

                                                //2.Split word into letters, count letters and reverse the string
                                                var wordSplit = word.split("");
                                                wordLength = word.length;

                                                for (i = wordLength - 1; i >= 0; i--) {
                                                    wordReverse += wordSplit[i];
                                                }

                                                if (word == wordReverse) {
                                                    $("#tacoCatOutput").text(word + " is a palindrome.");
                                                }
                                                else {
                                                    $("#tacoCatOutput").text(word + " is not a palindrome.");
                                                }
